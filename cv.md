## Code example

```javascript
function numberOfPairs(gloves) {
    let res = 0;
    const colors = Array.from(new Set(gloves));
    
    for (let i = 0; i < colors.length; i++) {
        const filter = gloves.filter((glove) => {
            return glove === colors[i];
        });

        res += Math.floor(filter.length / 2);
    }

    return res;
}
```
